class HelloWorld
 {
    public static void main(String[] args) 
     {
       
        int n=5;
        if(n%2==0)
        {
          System.out.println(n+"\t"+"is"+"\t"+"Even number");
        }
        else
        {
            System.out.println(n+"\t"+"is"+"\t"+"Odd number");
        }
    }
}


Output:

5 is Odd number
